282 ArrayList<T>& operator= (const ArrayList<T>& other)
283 {
284 // Save away all data we need from other
285 unsigned char* other_data_copy = new unsigned char[other.mCapacity * sizeof(T)];
286 memcpy(other_data_copy, other.mData, sizeof(T) * other.mCapacity);
287 unsigned int other_size = other.size();

293

300 // b = this_array_list = a;

591 T remove(unsigned int index, bool resize_if_necessary = true)
592 {
593 if (index >= mSize || index < 0)
594 throw std::out_of_range("Invalid index: " + std::to_string(index));
595
596 // Get the value we'll return at the end (the element removed)
597 T result = (T&)(mData[index * sizeof(T)]);
598
599 // Move all elements that come after index down one spot
600 memcpy(&mData[index * sizeof(T)], &mData[(index + 1) * sizeof(T)], (mSize - index - 1) * sizeof(T));
601
602 // Decrement our size
603 mSize--;
604
605 // Shrink, if applicable and requested
606 if (resize_if_necessary)
607 shrink();
608
609 // Return the result
610 return result;
611 }

642 {

708 void grow()
709 {
710 if (mSize == mCapacity)
711 {
712 // Allocate what will become the new array
713 unsigned char* new_array = nullptr;
714 if (mCapacity == 0)
715 {
716 new_array = new unsigned char[msMinCapacity * sizeof(T)];
717 memset(new_array, 0, msMinCapacity * sizeof(T));
718 }
719 else
720 {
721 new_array = new unsigned char[(mCapacity * 2) * sizeof(T)];
722 memset(new_array, 0, (mCapacity * 2) * sizeof(T));
723 }
724
725 // Copy over data from the old array (if any)
726 if (mData != nullptr)
727 {
728 memcpy(new_array, mData, sizeof(T) * mSize);
729
730 // Destroy the old array
731 delete[] mData;
732 }
733
734 // Make the new array *the* array
735 mData = new_array;
736
737 // Note that our capacity is now double what it used to be
738 if (mCapacity == 0)
739 mCapacity = msMinCapacity;
740 else
741 mCapacity *= 2;
742 }
